\documentclass[border=1mm,12pt]{standalone}
\usepackage{dcolumn}
\usepackage{tikz}
\usetikzlibrary{positioning}
\usepackage{fp}
\usepackage{siunitx}

% Define the number of decimal places
\def\decimals{3}
% Modify values
\def\decA{5.678}
\def\decB{123.4}

\sisetup{output-decimal-marker = {.}, round-mode=places, round-precision=\decimals}

\FPadd\decAB{\decA}{\decB}
\FPround\decAB{\decAB}{\decimals} % Round dp

\begin{document}

\begin{tabular}{c r}
    & \num{\decA} \\
    + & \num{\decB} \\
    \hline
    & \num{\decAB} \\
    \hline
\end{tabular}

\end{document}
