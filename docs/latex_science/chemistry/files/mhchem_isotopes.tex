\section*{Examples of isotope equations involving \\alpha decay, beta decay, and gamma radiation}

\begin{enumerate}
  \item Alpha decay: \ce{^{237}_{93}Np -> ^{233}_{91}Pa + ^{4}_{2}He}
  \item Alpha decay: \ce{^{238}_{92}U -> ^{234}_{90}Th + ^{4}_{2}He}
  \item Alpha decay: \ce{^{235}_{92}U -> ^{231}_{90}Th + ^{4}_{2}He}
  \item Alpha decay: \ce{^{226}_{88}Ra -> ^{222}_{86}Rn + ^{4}_{2}He}
  \item Alpha decay: \ce{^{210}_{84}Po -> ^{206}_{82}Pb + ^{4}_{2}He}
  \item Beta decay: \ce{^{14}_6C -> ^{14}_7N + \beta^- + \bar{\nu}_e}
  \item Beta decay: \ce{^{210}_{82}Pb -> ^{210}_{83}Bi + \beta^- + \bar{\nu}_e}
  \item Beta decay: \ce{^{3}_1H -> ^{3}_2He + \beta^- + \bar{\nu}_e}
  \item Beta decay: \ce{^{234}_{90}Th -> ^{234}_{91}Pa + \beta^- + \bar{\nu}_e}
  \item Beta decay: \ce{^{131}_{53}I -> ^{131}_{54}Xe + \beta^- + \bar{\nu}_e}
  \item Positron decay: \ce{^{11}_6C -> ^{11}_5B + \beta^+ + \nu_e}
  \item Positron decay: \ce{^{13}_7N -> ^{13}_6C + \beta^+ + \nu_e}
  \item Positron decay: \ce{^{15}_8O -> ^{15}_7N + \beta^+ + \nu_e}
  \item Positron decay: \ce{^{18}_9F -> ^{18}_8O + \beta^+ + \nu_e}
  \item Positron decay: \ce{^{22}_{11}Na -> ^{22}_{10}Ne + \beta^+ + \nu_e}
  \item Gamma decay: \ce{^{60}_{27}Co -> ^{60}_{27}Co + \gamma}
  \item Gamma decay: \ce{^{137}_{55}Cs -> ^{137}_{55}Cs + \gamma}
  \item Gamma decay: \ce{^{131}_{53}I -> ^{131}_{53}I + \gamma}
  \item Gamma decay: \ce{^{226}_{88}Ra -> ^{226}_{88}Ra + \gamma}
  \item Gamma decay: \ce{^{234}_{90}Th -> ^{234}_{90}Th + \gamma}
\end {enumerate}
