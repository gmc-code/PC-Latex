  \item Beta decay: \ce{^{131}_{53}I -> ^{\color{lightgray}\dots}_{\color{lightgray}\dots}Xe + \beta^- + \bar{\nu}_e}

  \item Gamma decay: \ce{^{60}_{27}Co -> ^{\color{lightgray}\dots}_{\color{lightgray}\dots}Co + \gamma}
  \item Gamma decay: \ce{^{137}_{55}Cs -> ^{\color{lightgray}\dots}_{\color{lightgray}\dots}Cs + \gamma}

\end{enumerate}
